package com.example.memoclient.views;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

import com.example.memoclient.R;
import com.example.memoclient.entities.Memo;
import com.example.memoclient.resources.MemoResource;

import retrofit2.Retrofit;
import retrofit2.Call;
import retrofit2.Callback;
import retrofit2.Response;
import retrofit2.converter.gson.GsonConverterFactory;

public class MemoActivity extends AppCompatActivity {

    private MemoResource memoResource;
    private TextView memoOutputText;
    private EditText memoInput;
    private Button saveButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_memo);

        memoOutputText = findViewById(R.id.output_text);
        memoInput = findViewById(R.id.input_text);
        saveButton = findViewById(R.id.save_button);

        memoResource = new Retrofit.Builder()
                .baseUrl("http://10.0.2.2:8080/")
                .addConverterFactory(GsonConverterFactory.create())
                .build()
                .create(MemoResource.class);

        restoreMemo();

        saveButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                String text = memoInput.getText().toString().trim();
                if (!text.isEmpty()) {
                    saveMemo(text);
                }
            }
        });
    }

    // GET
    private void restoreMemo() {
        memoResource.getMemo().enqueue(new Callback<Memo>() {
            @Override
            public void onResponse(Call<Memo> call, Response<Memo> response) {
                if (response.isSuccessful() && response.body() != null) {
                    memoOutputText.setText(response.body().getText());
                }
            }

            @Override
            public void onFailure(Call<Memo> call, Throwable t) {
                memoOutputText.setText("é€šä¿¡ã‚¨ãƒ©ãƒ¼");
            }
        });
    }

    // PUT
    private void saveMemo(String text) {
        memoResource.putMemo(text).enqueue(new Callback<Void>() {
            @Override
            public void onResponse(Call<Void> call, Response<Void> response) {
                memoInput.setText("");
                restoreMemo();
            }

            @Override
            public void onFailure(Call<Void> call, Throwable t) {
            }
        });
    }
}
